方形パルスのフーリエ変換

\begin{equation} f(x)=\binom{1(\lvert x \rvert \lt \frac{d/2})}{0(\lvert x \rvert \gt \frac{d/2})} \end{equation} 


import matplotlib.pyplot as plt
import numpy as np

fig = plt.figure(figsize=(8,6), dpi=100)
ax = fig.add_subplot(111)

d = 10
x = np.linspace(-10,10,100)
y = d * ((np.sin(x))/x)
plt.plot(x,y, linewidth=2, color='black')

ax.grid(True)
plt.show()
